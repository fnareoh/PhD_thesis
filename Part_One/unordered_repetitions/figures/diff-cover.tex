\newcommand{\deltacover}[2]{% (start_x, start_y) nb small block
\node (p) at #1 {};
\draw (p) rectangle ++(#2,-0.5); %draw Delta block
% draw multiples of sqrt(Delta)
\foreach \i in {1,...,#2} {
	\draw ($(p)+(\i,0)$) -- ($(p)+(\i,-0.5)$);
}
% dense region
\foreach \i in {1,...,8} {
	\draw ($(p)+(\i*0.125,0)$) -- ($(p)+(\i*0.125,-0.5)$);
}
}
